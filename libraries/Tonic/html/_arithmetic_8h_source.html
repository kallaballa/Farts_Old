226  inline void Multiplier_::computeSynthesisBlock( const SynthesisContext_ & context ){
227 
228  memset(&outputFrames_[0], 0, sizeof(TonicFloat) * outputFrames_.size());
229 
230  // for the first generator, store the value in the block
231  inputs_[0].tick(outputFrames_, context);
232 
233  // for additional generators, use the workspace stkframes for tick, and multiply it into the frames argument
234  for(int i = 1; i < inputs_.size(); i++) {
235  inputs_[i].tick(workSpace_, context);
236  outputFrames_ *= workSpace_;
237  }
238 
239  }
