20  void checkCAError(OSStatus error, const char *operation){
21  if (error == noErr) return;
22  char errorString[20];
23  // See if it appears to be a 4-char-code
24  *(UInt32 *)(errorString + 1) = CFSwapInt32HostToBig(error);

26  isprint(errorString[3]) && isprint(errorString[4])) {

28  errorString[6] = '\0';

30  // No, format it as an integer
