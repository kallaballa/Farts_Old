127  inline void FilteredFBCombFilter6_::computeSynthesisBlock( const SynthesisContext_ &context ){
128 
129  // tick modulations
130  delayTimeGen_.tick(delayTimeFrames_, context);
131 
132  TonicFloat y = 0;
133  TonicFloat * inptr = &dryFrames_[0];
134  TonicFloat * outptr = &outputFrames_[0];
135  TonicFloat * dtptr = &delayTimeFrames_[0];
136 
137  TonicFloat sf = scaleFactorCtrlGen_.tick(context).value;
138 
139  TonicFloat lowCoef = cutoffToOnePoleCoef(lowCutoffGen_.tick(context).value);
140  TonicFloat hiCoef = 1.0f - cutoffToOnePoleCoef(highCutoffGen_.tick(context).value);
141 
142  for (unsigned int i=0; i<kSynthesisBlockSize; i++){
143  onePoleLPFTick(delayLine_.tickOut(*dtptr++), lastOutLow_, lowCoef);
144  onePoleHPFTick(lastOutLow_, lastOutHigh_, hiCoef);
145  y = ((lastOutHigh_ * sf) + *inptr++); // no normalization on purpose
146  delayLine_.tickIn(y);
147  *outptr++ = y;
148  delayLine_.advance();
149  }
150 
151  }
