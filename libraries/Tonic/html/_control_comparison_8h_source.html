53  inline void ControlComparisonOperator_::computeOutput(const SynthesisContext_ & context){
54 
55  output_.triggered = false;
56 
57  ControlGeneratorOutput lhsOut = lhsGen_.tick(context);
58  ControlGeneratorOutput rhsOut = rhsGen_.tick(context);
59  if (lhsOut.triggered){
60  output_.value = satisfiesCondition(lhsOut.value, rhsOut.value) ? 1.0f : 0.0f;
61  output_.triggered = true;
62  }
63  else{
64  output_.value = 0;
65  }
66 
67  }
