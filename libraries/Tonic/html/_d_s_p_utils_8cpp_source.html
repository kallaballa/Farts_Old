17  // Discrete Fourier Transform
18  void DFT(int length, float *realTimeIn, float *imagTimeIn, float *realFreqOut, float *imagFreqOut)
19  {
20 
21  int k, i;
22  float sr, si, p;
23 
24  memset(realFreqOut, 0, sizeof(float) * length);
25  memset(imagFreqOut, 0, sizeof(float) * length);
26 
27  for(k = 0; k < length; k++)
28  for(i = 0; i < length; i++)
29  {
30  p = (2.0f * PI * (float)(k * i)) / length;
31  sr = cosf(p);
32  si = -sinf(p);
33  realFreqOut[k] += (realTimeIn[i] * sr) - (imagTimeIn[i] * si);
34  imagFreqOut[k] += (realTimeIn[i] * si) + (imagTimeIn[i] * sr);
35  }
36  }
