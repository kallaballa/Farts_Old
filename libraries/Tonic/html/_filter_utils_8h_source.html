53  inline static void bltCoef( TonicFloat b2, TonicFloat b1, TonicFloat b0, TonicFloat a1, TonicFloat a0, TonicFloat fc, TonicFloat *coef_out)
54  {
55  TonicFloat sf = 1.0f/tanf(PI*fc/Tonic::sampleRate());
56  TonicFloat sfsq = sf*sf;
57  TonicFloat norm = a0 + a1*sf + sfsq;
58  coef_out[0] = (b0 + b1*sf + b2*sfsq)/norm;
59  coef_out[1] = 2.0f * (b0 - b2*sfsq)/norm;
60  coef_out[2] = (b0 - b1*sf + b2*sfsq)/norm;
61  coef_out[3] = 2.0f * (a0 - sfsq)/norm;
62  coef_out[4] = (a0 - a1*sf + sfsq)/norm;
63  }
