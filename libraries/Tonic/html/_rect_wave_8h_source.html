55  inline void RectWave_::computeSynthesisBlock(const SynthesisContext_ &context){
56 
57  // tick freq and pwm
58  freqGen_.tick(freqFrames_, context);
59  pwmGen_.tick(pwmFrames_, context);
60 
61  const TonicFloat rateConstant = TONIC_RECT_RES / Tonic::sampleRate();
62 
63  TonicFloat *outptr = &outputFrames_[0];
64  TonicFloat *freqptr = &freqFrames_[0];
65  TonicFloat *pwmptr = &pwmFrames_[0];
66 
67  FastPhasor sd;
68 
69  // pre-multiply rate constant for speed
70 #ifdef USE_APPLE_ACCELERATE
71  vDSP_vsmul(freqptr, 1, &rateConstant, freqptr, 1, kSynthesisBlockSize);
72 #else
73  for (unsigned int i=0; i<kSynthesisBlockSize; i++){
74  *freqptr++ *= rateConstant;
75  }
76  freqptr = &freqFrames_[0];
77 #endif
78 
79  sd.d = BIT32DECPT;
80  TonicInt32 offs, msbi = sd.i[1];
81  double ps = phaseAccum_ + BIT32DECPT;
82  for ( unsigned int i=0; i<outputFrames_.frames(); i++ ) {
83 
84  sd.d = ps;
85  ps += *freqptr++;
86  offs = sd.i[1] & (TONIC_RECT_RES-1);
87  sd.i[1] = msbi;
88 
89  *outptr++ = offs > (TONIC_RECT_RES * *pwmptr++) ? -1.0f : 1.0f;
90  }
91 
92  sd.d = BIT32DECPT * TONIC_RECT_RES;
93  msbi = sd.i[1];
94  sd.d = ps + (BIT32DECPT * TONIC_RECT_RES - BIT32DECPT);
95  sd.i[1] = msbi;
96  phaseAccum_ = sd.d - BIT32DECPT * TONIC_RECT_RES;
97 
98  }

123  inline void RectWaveBL_::computeSynthesisBlock(const Tonic_::SynthesisContext_ &context)
124  {
125 
126  static const TonicFloat rateConstant = 1.0f / Tonic::sampleRate();
127 
128  // tick freq and pwm
129  freqGen_.tick(freqFrames_, context);
130  pwmGen_.tick(pwmFrames_, context);
131 
132  TonicFloat *outptr = &outputFrames_[0];
133  TonicFloat *freqptr = &freqFrames_[0];
134  TonicFloat *pwmptr = &pwmFrames_[0];
135 
136  // pre-multiply rate constant for speed
137 #ifdef USE_APPLE_ACCELERATE
138  vDSP_vsmul(freqptr, 1, &rateConstant, freqptr, 1, kSynthesisBlockSize);
139 #else

141  *freqptr++ *= rateConstant;
142  }
143  freqptr = &freqFrames_[0];
144 #endif
145 
146  // TODO: Maybe do this using a fast phasor for wraparound speed
147  for (unsigned int i=0; i<kSynthesisBlockSize; i++, pwmptr++, freqptr++, outptr++){
148 
149  phase_ += *freqptr;
150 
151  // add BLEP at end
152  if (phase_ >= 1.0)
153  {
154  phase_ -= 1.0;
155  accum_ = 0.0f;
156  addBLEP(phase_/(*freqptr), 1.0f);
157  }
158 
159  // add BLEP when we exceed pwm
160  if (accum_ == 0.f && phase_ > *pwmptr)
161  {
162  accum_ = 1.0f;
163  addBLEP((phase_ - (*pwmptr))/(*freqptr),-1.0f);
164  }
165 
166  *outptr = accum_;
167 
168  // add BLEP buffer contents
169  if (nInit_ > 0)
170  {
171  *outptr += ringBuf_[iBuffer_];
172  nInit_--;
173  if (++iBuffer_ >= lBuffer_) iBuffer_ = 0;
174  }
175 
176  // remove DC offset
177  *outptr = (*outptr * 2.f) - 1.f;
178 
179  }
