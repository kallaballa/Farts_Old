115  void Reverb_:: updateDelayTimes(const SynthesisContext_ & context)
116  {
117  ControlGeneratorOutput densityOutput = densityCtrlGen_.tick(context);
118  ControlGeneratorOutput shapeOutput = roomShapeCtrlGen_.tick(context);
119  ControlGeneratorOutput sizeOutput = roomSizeCtrlGen_.tick(context);
120  ControlGeneratorOutput decayOutput = decayTimeCtrlGen_.tick(context);
121 
122  if (densityOutput.triggered ||
123  shapeOutput.triggered ||
124  sizeOutput.triggered)
125  {
126 
127  reflectTapTimes_.clear();
128  reflectTapScale_.clear();
129 
130  // compute base round-trip times from listener to wall, based on shape and size values
131  TonicFloat shape = clamp(shapeOutput.value, 0.f, 1.f);
132  TonicFloat size = clamp(sizeOutput.value, 0.f, 1.f);
133 
134  TonicFloat wDist1 = map(size, 0.f, 1.f, TONIC_REVERB_MIN_WALL_DIST, TONIC_REVERB_MAX_WALL_DIST, true);
135  TonicFloat wDist2 = map(size * (1.1f - shape), 0.f, 1.f, TONIC_REVERB_MIN_WALL_DIST, TONIC_REVERB_MAX_WALL_DIST, true);
136 
137  unsigned int nTaps = (unsigned int)map(densityOutput.value, 0.f, 1.f, TONIC_REVERB_MIN_TAPS, TONIC_REVERB_MAX_TAPS, true);
138  TonicFloat tapScale = 1.0f/max(2.f, sqrtf(nTaps));
139  for (unsigned int i=0; i<nTaps; i++){
140 
141  TonicFloat dist = (i % 2 == 0 ? wDist1 : wDist2) * (1.0f + randomFloat(-TONIC_REVERB_FUDGE_AMT, TONIC_REVERB_FUDGE_AMT));
142 
143  reflectTapTimes_.push_back( dist/TONIC_REVERB_SOS );
144  reflectTapScale_.push_back( dBToLin(dist * TONIC_REVERB_AIRDECAY)*tapScale );
145  }
146 
147  }
148 
149  // if decay or room size have changed, need to update comb times/scales
150  if (sizeOutput.triggered ||
151  decayOutput.triggered)
152  {
153 
154  TonicFloat decayTime = decayOutput.value;
155  TonicFloat baseCombDelayTime = map(sizeOutput.value, 0.f, 1.f, TONIC_REVERB_MIN_COMB_TIME, TONIC_REVERB_MAX_COMB_TIME, true);
156 
157  for (unsigned int i=0; i<TONIC_REVERB_N_COMBS; i++){
158 
159  TonicFloat scaledDelayTime = combTimeScales_[i % TONIC_REVERB_N_COMBS] * baseCombDelayTime;
160  combFilterDelayTimes_[TONIC_LEFT][i].value(scaledDelayTime);
161  combFilterDelayTimes_[TONIC_RIGHT][i].value(scaledDelayTime+TONIC_REVERB_STEREO_SPREAD);
162  combFilterScaleFactors_[TONIC_LEFT][i].value( powf(10.f, (-3.0f * scaledDelayTime / decayTime)) );
163  combFilterScaleFactors_[TONIC_RIGHT][i].value( powf(10.f, (-3.0f * (scaledDelayTime+TONIC_REVERB_STEREO_SPREAD) / decayTime)) );
164 
165  }
166  }
167 
168  }
