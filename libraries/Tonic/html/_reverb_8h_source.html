47  inline void ImpulseDiffuserAllpass::tickThrough(Tonic::TonicFrames &frames)
48  {
49  TonicFloat *dptr = &frames[0];
50  TonicFloat y;
51  for (int i=0; i<kSynthesisBlockSize; i++){
52 
53  // feedback stage
54  y = *dptr + delayBack_.tickOut(delay_) * coef_;
55  delayBack_.tickIn(y);
56  delayBack_.advance();
57 
58  // feed forward stage
59  *dptr++ = (1.f+coef_)*delayForward_.tickOut(delay_) - y;
60  delayForward_.tickIn(y);
61  delayForward_.advance();
62  }
63  }

140  inline void Reverb_::computeSynthesisBlock(const SynthesisContext_ &context){
141 
142  updateDelayTimes(context);
143 
144  // pass thru input filters
145  if (inputFiltBypasCtrlGen_.tick(context).value == 0.f){
146 
147  inputLPF_.tickThrough(dryFrames_, workspaceFrames_[0], context);
148  inputHPF_.tickThrough(workspaceFrames_[0], workspaceFrames_[0], context);
149 
150  }
151  else{
152  workspaceFrames_[0].copy(dryFrames_);
153  }
154 
155  TonicFloat *wkptr0 = &(workspaceFrames_[0])[0];
156  TonicFloat *wkptr1 = &(workspaceFrames_[1])[0];
157 
158  // pass thru pre-delay, input filters, and sum the early reflections
159 
160  TonicFloat preDelayTime = preDelayTimeCtrlGen_.tick(context).value;
161 
162  for (unsigned int i=0; i<kSynthesisBlockSize; i++){
163 
164  // filtered input is in w0
165  // predelay output is in w1
166 
167  // pre-delay
168  preDelayLine_.tickIn(*wkptr0);
169  *wkptr1 = preDelayLine_.tickOut(preDelayTime);
170  preDelayLine_.advance();
171 
172  // taps - write back to w0
173  reflectDelayLine_.tickIn(*wkptr1++);
174  *wkptr0 = 0;
175  for (unsigned int t=0; t<reflectTapTimes_.size(); t++){
176  *wkptr0 += reflectDelayLine_.tickOut(reflectTapTimes_[t]) * reflectTapScale_[t];
177  }
178 
179  reflectDelayLine_.advance();
180  wkptr0++;
181  }
182 
183  // Comb filers
184  preOutputFrames_[TONIC_LEFT].clear();
185  preOutputFrames_[TONIC_RIGHT].clear();
186  for (unsigned int i=0; i<combFilters_[TONIC_LEFT].size(); i++){
187  combFilters_[TONIC_LEFT][i].tickThrough(workspaceFrames_[0], workspaceFrames_[1], context);
188  preOutputFrames_[TONIC_LEFT] += workspaceFrames_[1];
189  combFilters_[TONIC_RIGHT][i].tickThrough(workspaceFrames_[0], workspaceFrames_[1], context);
190  preOutputFrames_[TONIC_RIGHT] += workspaceFrames_[1];
191  }
192 
193  // Allpass filters
194  for (unsigned int i=0; i<allpassFilters_[TONIC_LEFT].size(); i++){
195  allpassFilters_[TONIC_LEFT][i].tickThrough(preOutputFrames_[TONIC_LEFT]);
196  allpassFilters_[TONIC_RIGHT][i].tickThrough(preOutputFrames_[TONIC_RIGHT]);
197  }
198 
199  // interleave pre-output frames into output frames
200  TonicFloat *outptr = &outputFrames_[0];
201  TonicFloat *preoutptrL = &preOutputFrames_[TONIC_LEFT][0];
202  TonicFloat *preoutptrR = &preOutputFrames_[TONIC_RIGHT][0];
203 
204  TonicFloat spreadValue = clamp(1.0f - stereoWidthCtrlGen_.tick(context).value, 0.f, 1.f);
205  TonicFloat normValue = (1.0f/(1.0f+spreadValue))*0.04f; // scale back levels quite a bit
206 
207  for (unsigned int i=0; i<kSynthesisBlockSize; i++){
208  *outptr++ = (*preoutptrL + (spreadValue * (*preoutptrR)))*normValue;
209  *outptr++ = (*preoutptrR++ + (spreadValue * (*preoutptrL++)))*normValue;
210  }
211 
212 
213  }
