54  inline void AngularWave_::computeSynthesisBlock(const SynthesisContext_ &context){
55 
56  // tick freq and slope inputs
57  freqGen_.tick(freqFrames_, context);
58  slopeGen_.tick(slopeFrames_, context);
59 
60  // calculate the output wave
61  TonicFloat const rateConstant = TONIC_SAW_RES/Tonic::sampleRate();
62 
63  TonicFloat slope, frac, phase;
64  TonicFloat *outptr = &outputFrames_[0];
65  TonicFloat *freqptr = &freqFrames_[0];
66  TonicFloat *slopeptr = &slopeFrames_[0];
67 
68  FastPhasor sd;
69 
70  // pre-multiply rate constant for speed
71 #ifdef USE_APPLE_ACCELERATE
72  vDSP_vsmul(freqptr, 1, &rateConstant, freqptr, 1, kSynthesisBlockSize);
73 #else
74  for (unsigned int i=0; i<kSynthesisBlockSize; i++){
75  *freqptr++ *= rateConstant;
76  }
77  freqptr = &freqFrames_[0];
78 #endif
79 
80  sd.d = BIT32DECPT;
81  TonicInt32 offs, msbi = sd.i[1];
82  double ps = phaseAccum_ + BIT32DECPT;
83  for ( unsigned int i=0; i<outputFrames_.frames(); i++ ) {
84 
85  // update the slope
86  slope = clamp(*slopeptr++, 0.0f, 1.0f) * TONIC_SAW_RES;
87 
88  sd.d = ps;
89  ps += *freqptr++;
90  offs = sd.i[1] & (TONIC_SAW_RES-1);
91  sd.i[1] = msbi;
92  frac = sd.d - BIT32DECPT;
93 
94  phase = offs + frac;
95 
96  // TODO: this is still a bit slow. Maybe find a way to do it with a real table lookup
97 
98  // on the rising edge
99  if (phase < slope){
100  *outptr++ = (phase/slope)*2.0f - 1.0f;
101  }
102  // on the falling edge
103  else{
104  *outptr++ = (1.0f - ((phase - slope)/(TONIC_SAW_RES - slope))) * 2.0f - 1.0f;
105  }
106 
107  }
108 
109  sd.d = BIT32DECPT * TONIC_SAW_RES;
110  msbi = sd.i[1];
111  sd.d = ps + (BIT32DECPT * TONIC_SAW_RES - BIT32DECPT);
112  sd.i[1] = msbi;
113  phaseAccum_ = sd.d - BIT32DECPT * TONIC_SAW_RES;
114 
115  }
