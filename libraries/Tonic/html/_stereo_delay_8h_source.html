64  TonicFloat *dryptr = &dryFrames_[0];

66  TonicFloat *fbkptr = &fbkFrames_[0];
67  TonicFloat *delptr_l = &(delayTimeFrames_[TONIC_LEFT])[0];

69 
70  for (unsigned int i=0; i<kSynthesisBlockSize; i++){

73  fbk = *fbkptr++;
74

76  outSamp[TONIC_RIGHT] = delayLine_[TONIC_RIGHT].tickOut(*delptr_r++);
77 
78  // output left sample

80  delayLine_[TONIC_LEFT].tickIn(*dryptr++ + outSamp[TONIC_LEFT] * fbk);
81

84  delayLine_[TONIC_RIGHT].tickIn(*dryptr++ + outSamp[TONIC_RIGHT] * fbk);
