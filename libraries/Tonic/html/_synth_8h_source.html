214  struct SynthFactory {
215 
216  typedef std::map<std::string, Synth(*)()> map_type;
217 
218  static Synth createInstance(std::string const& s) {
219  map_type::iterator it = getMap()->find(s);
220  if(it == getMap()->end()){
221  string synthsList = "";
222 
223  for(it = getMap()->begin(); it != getMap()->end(); it++){
224  synthsList = synthsList + it->first + "\n";
225  }
226 
227  error("Error creating synth. Synth named \"" + s + "\" not found. Existing registered synths are: \n" + synthsList);
228 
229  // return empty synth
230  return Synth();
231  }
232  return it->second();
233  }
234 
235  protected:
236  static map_type * getMap() {
237  // never delete'ed. (exist until program termination)
238  // because we can't guarantee correct destruction order
239  if(!map) { map = new map_type; }
240  return map;
241  }
242 
243  private:
244  static map_type * map;
245  };
