64  inline void TableLookupOsc_::computeSynthesisBlock( const SynthesisContext_ & context ){
65 
66  // Update the frequency data
67  frequencyGenerator_.tick(modFrames_, context);
68 
69  unsigned long tableSize = lookupTable_.size()-1;
70 
71  const TonicFloat rateConstant = (TonicFloat)tableSize / Tonic::sampleRate();
72 
73  TonicFloat *samples = &outputFrames_[0];
74  TonicFloat *rateBuffer = &modFrames_[0];
75  TonicFloat *tableData = lookupTable_.dataPointer();
76 
77  // R. Hoelderich style fast phasor.
78 
79  FastPhasor sd;
80 
81  // pre-multiply rate constant for speed
82 #ifdef USE_APPLE_ACCELERATE
83  vDSP_vsmul(rateBuffer, 1, &rateConstant, rateBuffer, 1, kSynthesisBlockSize);
84 #else
85  for (unsigned int i=0; i<kSynthesisBlockSize; i++){
86  *rateBuffer++ *= rateConstant;
87  }
88  rateBuffer = &modFrames_[0];
89 #endif
90 
91  sd.d = BIT32DECPT;
92 
93  TonicInt32 offs;
94  TonicInt32 msbi = sd.i[1];
95 
96  double frac;
97  double ps = phase_ + BIT32DECPT;
98 
99  TonicFloat *tAddr, f1, f2;
100 
101  for ( unsigned int i=0; i<kSynthesisBlockSize; i++ ) {
102 
103  sd.d = ps;
104  ps += *rateBuffer++;
105  offs = sd.i[1] & (tableSize-1);
106  tAddr = tableData + offs;
107  sd.i[1] = msbi;
108  frac = sd.d - BIT32DECPT;
109  f1 = tAddr[0];
110  f2 = tAddr[1];
111 
112  *samples++ = f1 + frac * (f2 - f1);
113  }
114 
115  sd.d = BIT32DECPT * tableSize;
116  msbi = sd.i[1];
117  sd.d = ps + (BIT32DECPT * tableSize - BIT32DECPT);
118  sd.i[1] = msbi;
119  phase_ = sd.d - BIT32DECPT * tableSize;
120 
121  }
122 
123  }
