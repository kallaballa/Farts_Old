239  inline bool isPowerOf2(unsigned int input, unsigned int * prevPo2){
240 
241  if (input == 0) return true;
242 
243  unsigned int po2 = 2;
244  while (po2 < input){
245  po2 *= 2;
246  }
247 
248  if (prevPo2){
249 
250  unsigned int nextPo2 = po2 * 2;
251  *prevPo2 = abs((int)input - (int)po2) < abs((int)input - (int)nextPo2) ? po2 : nextPo2;
252  }
253 
254  return input == po2;
255  }
