143 void TonicFrames :: resample( size_t nFrames , unsigned int nChannels )
144  {
145  if (nChannels > 2){
146  error("Invalid number of channels. TonicFrames is limited to mono or stereo only (1 or 2 channels)", true);
147  }
148 
149  if (nFrames != nFrames_ || nChannels != nChannels_){
150 
151 
152  // preserve as much of old data as we can
153  TonicFloat * oldData = data_;
154  unsigned long oldFrames = nFrames_;
155  unsigned int oldchannels = nChannels_;
156 
157  nFrames_ = nFrames;
158  nChannels_ = nChannels;
159 
160  size_ = nFrames_ * nChannels_;
161 
162  data_ = (TonicFloat *) malloc( size_ * sizeof( TonicFloat ) );
163 
164 #if defined(TONIC_DEBUG)
165  if ( data_ == NULL ) {
166  std::string error = "TonicFrames::resize: memory allocation error!";
167  Tonic::error(error, true);
168  }
169 #endif
170 
171  // resample the content (brute-force, no AA applied)
172  if (oldData){
173 
174  float inc = (float)oldFrames/nFrames_;
175 
176  for (unsigned int c=0; c<nChannels_; c++){
177 
178  float fIdx = 0.f;
179 
180  for (unsigned int i=0; i<nFrames_; i++){
181 
182  float fi, y1, y2;
183  float frac = modff(fIdx, &fi);
184  unsigned int idx = (unsigned int)fi;
185 
186  unsigned int ocIdx = oldchannels > 1 ? c : 0;
187 
188  if (idx == oldFrames-1){
189  data_[i*nChannels_ + c] = oldData[idx * oldchannels + ocIdx];
190  }
191  else{
192  y1 = oldData[idx * oldchannels + ocIdx];
193  y2 = oldData[(idx+1) * oldchannels + ocIdx];
194  data_[i*nChannels + c] = y1 + frac * (y2 - y1);
195  }
196 
197  // handle different channel mapping
198  if (oldchannels > nChannels_){
199  // add and average
200  y1 = oldData[idx * oldchannels + ocIdx + 1];
201  y2 = oldData[(idx+1) * oldchannels + ocIdx + 1];
202  data_[i*nChannels + c] += (y1 + frac * (y2 - y1));
203  data_[i*nChannels + c] *= 0.5f;
204  }
205 
206  fIdx += inc;
207  }
208  }
209  }
210 
211  bufferSize_ = size_;
212 
213  if (oldData) free(oldData);
214 
215  }

220 {
