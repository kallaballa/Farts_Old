273  inline void TonicFrames::copy( TonicFrames &f ){
274 
275 #if defined(TONIC_DEBUG)
276  if ( f.frames() != nFrames_) {
277  std::ostringstream error;
278  error << "TonicFrames::operator+=: frames argument must be of equal dimensions!";
279  Tonic::error(error.str(), true);
280  }
281 #endif
282 
283  unsigned int fChannels = f.channels();
284  TonicFloat *dptr = data_;
285  TonicFloat *fptr = &f[0];
286 
287  if (nChannels_ == fChannels){
288  memcpy(dptr, fptr, size_ * sizeof(TonicFloat));
289  }
290  else if (nChannels_ < fChannels){
291 
292  // sum channels
293  memset(dptr, 0, size_ * sizeof(TonicFloat));
294  for (unsigned int c=0; c<fChannels; c++){
295  dptr = data_;
296  fptr = &f(0,c);
297  for (unsigned int i=0; i<nFrames_; i++, dptr+=nChannels_, fptr+=fChannels){
298  *dptr += *fptr;
299  }
300  }
301 
302  // apply scaling (average of channels)
303  TonicFloat s = 1.0f/fChannels;
304  dptr = data_;
305  for (unsigned int i=0; i<nFrames_; i++, dptr+=nChannels_){
306  *dptr *= s;
307  }
308  }
309  else{
310  // just copy one channel, then fill
311  vcopy(dptr, nChannels_, fptr, fChannels, (unsigned int)nFrames_);
312 
313  // fill all channels if necessary
314  fillChannels();
315  }
316
